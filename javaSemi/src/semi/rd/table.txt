
==================== 테이블 생성 쿼리 ====================

CREATE TABLE Researcher (
	 researcher_code VARCHAR2(10) PRIMARY KEY,
	 org_code VARCHAR2(10) NOT NULL,
	 name VARCHAR2(50) NOT NULL,
	 tel VARCHAR2(50) NOT NULL,
	 email VARCHAR2(100) NULL
);

CREATE TABLE Researcher_Role (
	 project_code VARCHAR2(10) NOT NULL,
	 researcher_code VARCHAR2(10) NOT NULL,
	 role VARCHAR2(50) NOT NULL,
	 start_date DATE NOT NULL,
	 end_date DATE NOT NULL,
	 PRIMARY KEY(project_code, researcher_code)
);

CREATE TABLE Project (
	 project_code VARCHAR2(10) NOT NULL PRIMARY KEY,
	 org_code VARCHAR2(10) NOT NULL,
	 title VARCHAR2(2000) NOT NULL,
	 stage VARCHAR2(50) NOT NULL,
	 status VARCHAR2(50) NOT NULL,
	 budget NUMBER NOT NULL,
	 start_Date DATE NOT NULL,
	 end_date DATE NOT NULL
);

CREATE TABLE Project_Ministry (
	 project_code VARCHAR2(10) NOT NULL,
	 ministry_code VARCHAR2(10) NOT NULL,
	 PRIMARY KEY(project_code, ministry_code)
);

CREATE TABLE Fund_Management (
	 fund_code VARCHAR2(10) NOT NULL PRIMARY KEY,
	 project_code VARCHAR2(10) NOT NULL,
	 researcher_code VARCHAR2(10) NOT NULL,
	 charger_name VARCHAR2(50) NOT NULL,
	 category VARCHAR2(50) NOT NULL,
	 date_used DATE NOT NULL,
	 expense NUMBER NOT NULL,
	 content CLOB NOT NULL,
	 vendor_name VARCHAR2(50) NULL,
	 proof_type VARCHAR2(50) NOT NULL,
	 memo VARCHAR2(2000) NULL
);

CREATE TABLE Organization (
	 org_code VARCHAR2(10) NOT NULL PRIMARY KEY,
	 name VARCHAR2(50) NOT NULL,
	 type VARCHAR2(50) NOT NULL,
	 biz_reg_no VARCHAR2(50) NOT NULL,
	 tel VARCHAR2(50) NOT NULL,
	 email VARCHAR2(100) NULL,
	 address VARCHAR2(200) NOT NULL,
	 id VARCHAR2(50) NOT NULL,
	 pwd VARCHAR2(50) NOT NULL
);

CREATE TABLE Milestone (
	 milestone_code VARCHAR2(10) PRIMARY KEY, --마일스톤 코드
	 project_code VARCHAR2(10) NOT NULL, --과제코드
	 name VARCHAR2(2000) NOT NULL, --마일스톤이름
	 description CLOB NOT NULL, --설명
	 p_end_date DATE NOT NULL, --계획완료일
	 a_end_date DATE NOT NULL, --실제완료일
	 status VARCHAR2(50) NOT NULL --상태
);

CREATE TABLE Performance_Management (
	 performance_code VARCHAR2(10) PRIMARY KEY, --성과관리코드
	 project_code VARCHAR2(10) NOT NULL, --과제코드
	 name VARCHAR2(2000) NOT NULL, --성과 이름
	 category VARCHAR2(50) NOT NULL, -- 성과관리분류
	 content CLOB NOT NULL, --성과관리 내용
	 p_date DATE NOT NULL, --성과발생 일자
	 memo VARCHAR2(2000) NULL --비고
);

CREATE TABLE Project_Partner_Org (
	 project_code VARCHAR2(10) NOT NULL, --과제코드
	 partner_org_code VARCHAR2(10) NOT NULL, --협약기관코드
	 PRIMARY KEY(project_code, partner_org_code)
);

CREATE TABLE Partner_Organization (
	 partner_org_code VARCHAR2(10) PRIMARY KEY, --협약기관코드
	 name VARCHAR2(50) NOT NULL, 
	 type VARCHAR2(50) NOT NULL,
	 biz_reg_no VARCHAR2(50) NOT NULL,
	 tel VARCHAR2(50) NOT NULL,
	 email VARCHAR2(100) NULL,
	 address VARCHAR2(50) NULL,
	 charger_name VARCHAR2(50) NOT NULL,
	 charger_tel VARCHAR2(50) NOT NULL,
	 bank_name VARCHAR2(50) NOT NULL,
	 account_number VARCHAR2(50) NOT NULL,
	 account_holder VARCHAR2(50) NOT NULL
);

CREATE TABLE Ministry (
	 ministry_code VARCHAR2(10) PRIMARY KEY,
	 name VARCHAR2(50) NOT NULL,
	 tel VARCHAR2(50) NOT NULL,
	 email VARCHAR2(100) NULL
);
